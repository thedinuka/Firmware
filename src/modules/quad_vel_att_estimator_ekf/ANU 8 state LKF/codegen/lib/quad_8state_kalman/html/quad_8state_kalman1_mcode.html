1function y = quad_8state_kalman(a,g, delta_t, gyro_rp, gyro_y, v_xy, gb_rp, gb_y, acc_xy, k_1) %#codegen
 2
 3%states {phi, theta, psi, vx vy, beta_gx, beta_gy, beta_gz}
 4
 5u = [g(1); g(2); g(3)];
 6z = [a(1); a(2)];
 7
 8
 9%continuous state transition matrix
10A = single(zeros(8,8));
11A(1,6) = -1;
12A(2,7) = -1;   
13A(3,8) = -1;
14A(4,2) = -9.81;
15A(4,4) = k_1;
16A(5,1) = 9.8;
17A(5,5) = k_1;
18
19%discretize state transition matrix
20A = eye(8)+ A.*delta_t;
21
22%continuous control matrix
23G = single(zeros(8,3));             
24G(1,1) = 1;
25G(2,2) = 1;
26G(3,3) = 1;
27
28B = A*G*delta_t;
29
30%measurement matrix
31H = single(zeros(2,8));
32H(1,4) = k_1;
33H(2,5) = k_1;
34
35%process noise matrix
36Q = diag([gyro_rp, gyro_rp, gyro_y, v_xy, v_xy, gb_rp, gb_rp, gb_y])*delta_t;
37
38%measurement noise matrix
39R = diag([acc_xy, acc_xy]);
40
41
42persistent p_est x_est
43%initialization
44if isempty(x_est)
45    p_est = single(diag([0,0,0,     0,0,     0.1, 0.1, 0.1]));
46    x_est = single([0,0,0,     0,0,    0,0,0]');
47end
48
49% Predicted state and covariance
50x_prd = A * x_est + B * u;
51p_prd = A * p_est * A' + Q;
52
53% Estimation
54S = H * p_prd' * H' + R;
55inv = H * p_prd';
56klm_gain = (S \ inv)';
57
58% Estimated state and covariance
59x_est = x_prd + klm_gain * (z - H * x_prd);
60p_est = p_prd - klm_gain * H * p_prd;
61
62y = x_est;
63
64end
